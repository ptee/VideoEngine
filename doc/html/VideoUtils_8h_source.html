146 int VideoUtils::getCodec(cv::VideoCapture* capture)
147 {
148  if ( capture == NULL )
149  return VideoDefs::NULL_VIDEO_CAPTURE;
150 
151  // Get Codec Type- Int form
152  // All the video codecs have a unique short name of maximum four characters.
153  // e.g. XVID, DIVX or H264 names.
154  // Four characters are four bytes, meaning 32 bits.
155  // These four characters are coded in the lower 32 bits of the double.
156  // A simple way to throw away the upper 32 bits would be to just convert this value to int:
157  return static_cast<int>(capture->get(CV_CAP_PROP_FOURCC));
158 }

162 QString VideoUtils::getCodecName(cv::VideoCapture* capture)
163 {
164  QString codec("");
165  int ex = getCodec(capture);
166  if ( ex == VideoDefs::NULL_VIDEO_CAPTURE )
167  return codec;
168 
169  // Transform from int to char via Bitwise operators
170  char EXT[] = {(char)(ex & 0XFF) , (char)((ex & 0XFF00) >> 8),(char)((ex & 0XFF0000) >> 16),(char)((ex & 0XFF000000) >> 24), 0};
171  codec = EXT;
172  return codec;
