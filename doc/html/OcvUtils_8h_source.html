86  void OcvUtils::ShiftDft(cv::Mat& src)
87  {
88  int cx = src.cols/2;
89  int cy = src.rows/2;
90 
91  // rearrange the quadrants of Fourier image
92  // so that the origin is at the image center
93  cv::Mat tmp;
94  // Rect started at bottom-left as (x,y), whereas cx, cy is its widtht and height.
95  cv::Mat q0(src, cv::Rect(0, 0, cx, cy));
96  cv::Mat q1(src, cv::Rect(cx, 0, cx, cy));
97  cv::Mat q2(src, cv::Rect(0, cy, cx, cy));
98  cv::Mat q3(src, cv::Rect(cx, cy, cx, cy));
99 
100  q0.copyTo(tmp);
101  q3.copyTo(q0);
102  tmp.copyTo(q3);
103 
104  q1.copyTo(tmp);
105  q2.copyTo(q1);
106  tmp.copyTo(q2);

130  bool OcvUtils::stretchImage(const cv::Mat_<T>& img, cv::Mat_<uchar>& out, int a_min, int a_max, double filterMin, double filterMax)
131  {
132  if ( a_min < 0 || a_max > 255 || filterMin < 1.0 || filterMax > 254.0 )
133  return false;
134 
135  double minV,maxV;
136  cv::minMaxLoc(img, &minV, &maxV);
137  double factor = (a_max - a_min)/(maxV - minV);
138  double minVal = minV+(factor*filterMin/100);
139  double maxVal = maxV-(factor*filterMax/100);
140  for(int i=0; i< img.rows; ++i) {
141  for(int j=0; j< img.cols; ++j) {
142  double v = factor*img(i,j);
143  v = v < minVal? a_min : v > maxVal? a_max: v;
144  out(i,j) = static_cast<uchar>( round(v) );
145  }
146  }
147  return true;
148  }
